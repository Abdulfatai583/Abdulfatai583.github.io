International FMCG SQL Queries - Sample

International FMCG Sales Analysis - SQL Queries

-- Q1 Total profit of breweries by territory (2017-2019)
SELECT CASE 
    WHEN COUNTRIES IN ('Nigeria','Ghana','Senegal','Togo','Benin') THEN 'Francophone'
    WHEN COUNTRIES IN ('Nigeria','Ghana','Senegal','Togo','Benin') THEN 'Anglophone'
    ELSE 'Other'
END AS TERRITORY,
SUM(PROFIT) AS TOTAL_PROFIT
FROM international_fmcg_sql_assessment
WHERE YEARS BETWEEN 2017 AND 2019
GROUP BY TERRITORY;

-- Q2 Compare total profit between territories
SELECT CASE 
    WHEN COUNTRIES IN ('Ghana','Nigeria','Togo','Senegal','Benin') THEN 'Francophone'
    WHEN COUNTRIES IN ('Ghana','Nigeria','Togo','Senegal','Benin') THEN 'Anglophone'
    ELSE 'Other'
END AS Territory,
SUM(PROFIT) AS Total_Profit
FROM international_fmcg_sql_assessment
GROUP BY CASE 
    WHEN COUNTRIES IN ('Ghana','Nigeria','Togo','Senegal','Benin') THEN 'Francophone'
    WHEN COUNTRIES IN ('Ghana','Nigeria','Togo','Senegal','Benin') THEN 'Anglophone'
    ELSE 'Other'
END;

-- Q3 Country with highest profit in 2019
SELECT COUNTRIES, SUM(PROFIT) AS Total_Profit
FROM international_fmcg_sql_assessment
WHERE YEARS = 2019
GROUP BY COUNTRIES
LIMIT 1;

-- Q4 Year with highest profit
SELECT YEARS, SUM(PROFIT) AS Total_Profit
FROM international_fmcg_sql_assessment
GROUP BY YEARS
ORDER BY Total_Profit DESC;

-- Q5 Month with least profit in three years
SELECT MONTHS, SUM(PROFIT) AS total_profit
FROM international_fmcg_sql_assessment
GROUP BY MONTHS
ORDER BY total_profit ASC
LIMIT 1;
